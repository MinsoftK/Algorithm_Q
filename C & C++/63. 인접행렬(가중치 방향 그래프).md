## 내가 짠 코드
* 생각보다 간단했다.

<br/>

```c++
#include <stdio.h>
#include <vector>
#include <algorithm>
#include <stack>
using namespace std;

int map[21][21];

int main()
{
//	freopen("input.txt", "rt", stdin);
	int i, j,  n, line;
	int tmp_i, tmp_j, len;
	scanf("%d %d", &n, &line);
	for(i = i; i <= line; i++)
	{
		scanf("%d %d %d", &tmp_i, &tmp_j, &len);
		map[tmp_i][tmp_j] = len;
	}
	for(i = 1; i <= n; i++)
	{
		for (j = 1; j <= n; j++)
		{
			printf("%d ", map[i][j]);
		}
		printf("\n");
	}
	return(0);
}

//51 다이나믹 
//56 재귀
//58 이진트리 깊이우선탐색 


```
