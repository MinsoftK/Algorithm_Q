```c++
#include <iostream>
#include <vector>

using namespace std;

int main()
{
//	freopen("input.txt", "rt", stdin);
	vector<int> a; // 빈 벡터 생성, a라는 벡터 객체 생성 
	a.push_back(6);
	a.push_back(8);
	a.push_back(11);
	cout<< a.size()<<endl; //벅테의 원소의 개수를 return 받는다. 
	
	vector<int> b;
	b[1] = 5;
//	cout<<b[1]<<endl; // 빈 벡터이기 때문에 출력이 안된다. 
//  vector<int> b(5) // 값이 0 인 5개의 벡터가 생성된다.


	vector<int> c[3]; // 벡터가 3개가 생성이 된다. c[0], c[1], c[2] 벡터 배열이다.
	c[0].push_back(1); 
	c[0].push_back(3);
	c[0].push_back(5);
	c[1].push_back(2);
	c[1].push_back(4);
	c[1].push_back(6);
	c[2].push_back(7);
	cout<<c[1][1]<<endl;
	cout<<c[2][0]<<endl;
	
	vector<pair<int, int> > g[3];		//{ a, b} 앞부분은 first고 뒷부분이 second다. 
	g[1].push_back({3,5});
	g[1].push_back({4,7});
	g[1].push_back({5,9}};
	g[2].push_back(make_pair(7,7));	
	cout<< g[2][0].first << " " << g[2][0].second << endl;
}
```
