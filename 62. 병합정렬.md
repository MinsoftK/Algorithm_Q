# 62. 병합정렬

* N개이 숫자가 입력되면 오름차순으로 정렬하여 출력하는 프로그램을 작성하세요.
정렬하는 방법은 병합정렬입니다.


<br/>

#### ▣ 입력설명

* 첫 번째 줄에 자연수 N(1<=N<=100)이 주어집니다.
두 번째 줄에 N개의 자연수가 공백을 사이에 두고 입력됩니다. 각 자연수는 정수형 범위 안에 
있습니다. 




#### ▣ 출력설명

* 오름차순으로 정렬된 수열을 출력합니다.

7 6 3 1 5 2 4 8

#include <stdio.h>
#include <vector>
#include <algorithm>
#include <stack>
using namespace std;
int data[10], tmp[10];

void divide(int idx, int num)
{
	int mid;
	int p1, p2, p3; 
	if (idx < num)
	{
		mid = (idx + num) / 2; //이분검색처럼 mid를 구한다.
		divide(idx, mid);	//왼쪽 자식 노드 
		divide(mid + 1, num); //오른쪽 자식 노드 
		p1 = idx;	 
		p2 = mid + 1;
		p3 = idx;
		while (p1 <= mid && p2 <= num)
		{
			if (data[p1] < data[p2])
				tmp[p3++] = data[p1++];
			else
				tmp[p3++] = data[p2++];
		}
		while (p1 <= mid)
			tmp[p3++] = data[p1++];
		while (p2 <= num)
			tmp[p3++] = data[p2++];
		for (int i = idx; i <= num; i++)
		{
			data[i] = tmp[i]; //각각의 노드에서 정렬된 값을 data 배열에 다시 저장한다. 
		}
	}
}

int main()
{
	freopen("input.txt", "rt", stdin);
	int n, i;
	scanf("%d", &n);
	for (i = 1; i <= n; i++)
	{
		scanf("%d", &data[i]);
	}
	divide(1, n);
	for (i = 1; i <=n; i++)
	{
		printf("%d ", data[i]);
	}
	return(0);
}
