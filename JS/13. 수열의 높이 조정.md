function solution(arr, m) {
	let max = Number.MIN_SAFE_INTEGER;
	let min = Number.MAX_SAFE_INTEGER;
	let ch = new Array(1000).fill(0);
	for (let i = 0; i < arr.length; i++) {
		ch[arr[i]] += 1;
		if (max < ch[arr[i]]) max = arr[i];
		if (min > ch[arr[i]]) min = arr[i];
	}
	for (let i = 1; i < m; i++) {
		if (max === min) return 0;
		if (ch[max] === 1) {
			ch[max]--;
			max--;
			ch[max]++;
		} else {
			ch[max]--;
			ch[max - 1]++;
		}
		if (ch[min] === 1) {
			ch[min]--;
			min++;
			ch[min]++;
		} else {
			ch[min]--;
			ch[min + 1]++;
		}
	}
	answer = max - min;
	return answer;
}
