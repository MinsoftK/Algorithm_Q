function solution(s) {
	let n = s.length;
	let left = 0,
		right = n - 1;
	let arr = Array(s.length).fill(0);
	for (let i = n - 1; i > 0; i--) {
		if (Math.abs(arr[left]) > Math.abs(arr[right])) {
			arr[i] = arr[left] * arr[left];
			left++;
		} else {
			arr[i] = arr[right] * arr[right];
			right--;
		}
	}
	console.log(arr);
}
console.log(solution([-4, -1, 0, 3, 10]));
